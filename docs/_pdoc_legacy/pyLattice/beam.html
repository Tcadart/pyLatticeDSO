 17    """
 18    Class Beam represents a beam element defined by two endpoints (Point objects), a radius, material index, type,
 19    and the cells it belongs to.

 22    def __init__(self, point1: 'Point', point2: 'Point', radius: float, material: int, type_beam: int,
 23                 cell_belongings: ['Cell' or list['Cell']]) -> None:
 24        """
 25        Initialize a Beam object representing a beam element.
 26
 27        Parameters:
 28        ----------
 29        point1 : Point
 30            The first endpoint of the beam.
 31
 32        point2 : Point
 33            The second endpoint of the beam.
 34
 35        radius : float
 36            The radius of the beam.
 37
 38        material : int
 39            The material index of the beam.
 40
 41        type_beam : int
 42            Reference to the geometry type of the beam in case of multiple geometries.
 43
 44        cell_belongings : Cell or list of Cell
 45            The cell(s) to which the beam belongs.
 46        """
 47        if point1 is point2:
 48            raise ValueError("A beam cannot have the same point for both endpoints.")
 49        if radius <= 0:
 50            raise ValueError("Radius must be a positive value.")
 51        if not isinstance(material, int) or material < 0:
 52            raise ValueError("material must be a non-negative integer.")
 53
 54        self.point1: 'Point' = point1
 55        self.point2: 'Point' = point2
 56        self.radius: float = radius
 57        self.material: int = material
 58        self.type_beam: int = type_beam
 59        if not isinstance(cell_belongings, list):
 60            self.cell_belongings: list['Cell'] = [cell_belongings]
 61        else:
 62            self.cell_belongings: list['Cell'] = cell_belongings
 63        self.length: float = self.get_length()
 64        self.volume: float = self.get_volume(section_type="circular")
 65
 66        # Attributes for simulation
 67        self.index: Optional[int] = None
 68        self.angle_point_1: dict = {"radius": None, "angle": None, "L_zone": None}
 69        self.angle_point_2: dict = {"radius": None, "angle": None, "L_zone": None}
 70        self.beam_mod: bool = False
 71        self.penalization_coefficient: float = 1.5  # Fixed (See publication Cadart et al. 2025)
 72        self.associated_beams_mod: List['Beam'] = []
 73        self.initial_radius: Optional[float] = None

 85    @timing.timeit
 86    def destroy(self) -> None:
 87        """
 88        Delete the beam and clean up references in points and cells.
 89        This method removes the beam from its associated cells and points.
 90        """
 91        # delete the beam from the cells
 92        for cell in list(self.cell_belongings):
 93            _discard(cell.beams_cell, self)
 94        self.cell_belongings.clear()
 95
 96        # delete the beam from the points
 97        for p in (self.point1, self.point2):
 98            cb = getattr(p, "connected_beams", None)
 99            _discard(cb, self)
100
101
102        # delete the point if it has no more connected beams
103        for p in (self.point1, self.point2):
104            cb = getattr(p, "connected_beams", None)
105            if not cb:
106                p.destroy()

127        Calculate the length of the beam.
128
129        Returns:

133        x1, y1, z1 = self.point1.x, self.point1.y, self.point1.z
134        x2, y2, z2 = self.point2.x, self.point2.y, self.point2.z
135        length = round(math.sqrt((x2 - x1) ** 2 + (y2 - y1) ** 2 + (z2 - z1) ** 2), 4)

159    def is_identical_to(self, other: "Beam", tol:float = 1e-9) -> bool:
160        """
161        Check if this beam is identical to another beam.
162
163        Parameters
164        ----------
165        other : Beam
166            The other beam to compare with.
167        tol : float
168            Tolerance for floating-point comparisons.
169        """
170        if not isinstance(other, Beam):
171            return False
172        if not math.isclose(self.radius, other.radius, rel_tol=1e-8, abs_tol=tol):
173            return False
174        if self.material != other.material or self.type_beam != other.type_beam:
175            return False
176
177        def _pt_close(a: "Point", b: "Point") -> bool:
178            return (math.isclose(a.x, b.x, abs_tol=tol) and
179                    math.isclose(a.y, b.y, abs_tol=tol) and
180                    math.isclose(a.z, b.z, abs_tol=tol))
181
182        same_order = _pt_close(self.point1, other.point1) and _pt_close(self.point2, other.point2)
183        swap_order = _pt_close(self.point1, other.point2) and _pt_close(self.point2, other.point1)
184        return same_order or swap_order

204    def get_angle_between_beams(self, other: 'Beam', periodicity: bool) -> float:
205        """
206        Calculates the angle between two beams
207
208        Parameters:
209        ----------
210        other : Beam
211            The other beam to calculate the angle with.
212
213        periodicity : bool
214            If True, considers periodic boundary conditions.
215
216        Return:
217        --------
218        Angle: float
219            angle in degrees
220        """
221        p1, p2 = None, None
222        if periodicity:
223            # Tag for corners (1000-1007)
224            for idx1, p1_candidate in enumerate([self.point1, self.point2]):
225                if p1_candidate.tag and 1000 <= p1_candidate.tag <= 1007:
226                    p1 = idx1
227                    for idx2, p2_candidate in enumerate([other.point1, other.point2]):
228                        if p2_candidate.tag and 1000 <= p2_candidate.tag <= 1007:
229                            p2 = idx2  # Enregistrer l'indice pour beam2
230                            break
231
232            # Tags for edges (100-111)
233            if p1 is None and p2 is None:
234                list_tag_edge = [[102, 104, 106, 107], [100, 108, 105, 111], [101, 109, 103, 110]]
235                for tag_list in list_tag_edge:
236                    for idx1, p1_candidate in enumerate([self.point1, self.point2]):
237                        if p1_candidate.tag and p1_candidate.tag in tag_list:
238                            p1 = idx1
239                            for idx2, p2_candidate in enumerate([other.point1, other.point2]):
240                                if p2_candidate.tag and p2_candidate.tag in tag_list:
241                                    p2 = idx2
242                                    break
243            # Tags for faces (10-15)
244            if p1 is None and p2 is None:
245                list_face_tag = [[10, 15], [11, 14], [12, 13]]
246                for face_tag in list_face_tag:
247                    for idx1, p1_candidate in enumerate([self.point1, self.point2]):
248                        if p1_candidate.tag and p1_candidate.tag in face_tag:
249                            p1 = idx1
250                            for idx2, p2_candidate in enumerate([other.point1, other.point2]):
251                                if p2_candidate.tag and p2_candidate.tag in face_tag:
252                                    p2 = idx2
253                                    break
254
255        if self.point1 is other.point1 or (p1 == 0 and p2 == 0):
256            u = self.point2 - self.point1
257            v = other.point2 - other.point1
258        elif self.point1 is other.point2 or (p1 == 0 and p2 == 1):
259            u = self.point2 - self.point1
260            v = other.point1 - other.point2
261        elif self.point2 is other.point1 or (p1 == 1 and p2 == 0):
262            u = self.point1 - self.point2
263            v = other.point2 - other.point1
264        elif self.point2 is other.point2 or (p1 == 1 and p2 == 1):
265            u = self.point1 - self.point2
266            v = other.point1 - other.point2
267        else:
268            raise ValueError("beams are not connected at any point")
269
270        dot_product = sum(a * b for a, b in zip(u, v))
271        u_norm = math.sqrt(sum(a * a for a in u))
272        v_norm = math.sqrt(sum(b * b for b in v))
273        cos_theta = dot_product / (u_norm * v_norm)
274        cos_theta = max(min(cos_theta, 1.0), -1.0)
275        angle_rad = math.acos(cos_theta)
276        angle_deg = math.degrees(angle_rad)
277        return angle_deg

279    def get_point_on_beam_at_distance(self, distance: float, start_point: int) -> "Point":
280        """
281        Calculate the coordinates of a point on the beam at a specific distance from an endpoint.
282
283        Parameters:
284        ----------
285        distance : float
286            The distance from the specified endpoint along the beam.
287
288        start_point : int
289            The index of the starting point (1 or 2) from which the distance is measured
290
291        Returns:
292        -------
293            List[float]: Coordinates [x, y, z] of the calculated point.
294
295        Raises:
296        -------
297            ValueError: If the point index is not 1 or 2.
298        """
299        if start_point == 1:
300            start_point = self.point1
301            end_point = self.point2
302        elif start_point == 2:
303            start_point = self.point2
304            end_point = self.point1
305        else:
306            raise ValueError("point must be 1 or 2.")
307
308        direction_ratios = [
309            (end_point.x - start_point.x) / self.length,
310            (end_point.y - start_point.y) / self.length,
311            (end_point.z - start_point.z) / self.length,
312        ]
313
314        factors = [dr * distance for dr in direction_ratios]
315
316        point_mod = [
317            start_point.x + factors[0],
318            start_point.y + factors[1],
319            start_point.z + factors[2]
320        ]
321
322        if self.type_beam == 2:
323            cell_belongings = self.cell_belongings
324        else:
325            cell_belongings = list(set(self.point1.cell_belongings) & set(self.point2.cell_belongings))
326        point_mod = Point(*point_mod, cell_belongings=cell_belongings)
327
328        return point_mod

331    @timing.timeit
332    def is_point_on_beam(self, node: 'Point') -> bool:
333        """
334        Check if a given node lies on the beam.
335
336        Parameters:
337        ----------
338        node : Point
339            The node to check.
340
341        Returns:
342        -------
343            bool: True if the node lies on the beam, False otherwise.
344        """
345        vector1 = (self.point2.x - self.point1.x, self.point2.y - self.point1.y, self.point2.z - self.point1.z)
346        vector2 = (node.x - self.point1.x, node.y - self.point1.y, node.z - self.point1.z)
347
348        if (node.x == self.point1.x and node.y == self.point1.y and node.z == self.point1.z) or (
349                node.x == self.point2.x and node.y == self.point2.y and node.z == self.point2.z):
350            return False
351        cross_product = (
352            vector1[1] * vector2[2] - vector1[2] * vector2[1],
353            vector1[2] * vector2[0] - vector1[0] * vector2[2],
354            vector1[0] * vector2[1] - vector1[1] * vector2[0]
355        )
356
357        if cross_product == (0, 0, 0):
358            dot_product = (vector2[0] * vector1[0] + vector2[1] * vector1[1] + vector2[2] * vector1[2])
359            vector1_length_squared = (vector1[0] ** 2 + vector1[1] ** 2 + vector1[2] ** 2)
360            return 0 <= dot_product <= vector1_length_squared
361        else:
362            return False
